\begin{theorem}
Let the following:
$\forall S \in \mathbb{R}, \, Baud(MIN) < Freq(S) < Baud(MAX).$ Where $Baud(MIN)$ is the minimal baud rate and $Baud(MAX)$ the maximal baud rate:
\begin{align*}
	Freq(S) &= \frac{Hi(S)}{Lo(S)} Weight(x),\\
	-\frac{Freq(S)}{Weight(S)} &= \frac{Hi(S)}{Lo(S)}.
\end{align*}
Per Hartley's law, the weight of $S$ over its frequency $F$, may be equal of the division between the highest signal and lowest signal of $S$.
\end{theorem}

\begin{remark}
	There exist degenerate cases for 2.1, per the weight $S$ over $F$. Where $S$ is greater than $F$.
\end{remark}
