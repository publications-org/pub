\section{DEFINITIONS}

We assume these definitions are valid within their conditions:

\subsection{DEFINITION OF $L(t) \in \mathbb{R}$}
We define:
\begin{equation}
	L(b, a) \coloneqq \int_{a}^{b}X(u, n)\cdot du, \quad n \coloneqq b - a, \quad n \in \mathbb{Z}^{0+}, \quad, \quad X(t, p) \coloneqq \sum_{k=1}^{p}\frac{t}{p}\cdot kt, \quad p \geq 1, \quad p \in \mathbb{Z}^{+}
\end{equation}
Where $\forall t \in \mathbb{R}, t > 1$.

\subsection{DEFINITION OF $W(t, a, b) \in \mathbb{R}$}
We define:
\begin{equation}
    W(b, a) \coloneqq L(b, a)\cdot(\sum_{k=a}^{b} L(b, k)) +C_{rem}, \quad k \neq b,\quad \forall C_{rem} \in \mathbb{R}, \quad C > 0
\end{equation}
Where $\forall t \in \mathbb{R}, t > 0$ and $C_{rem}$ is a remainder constant.

\subsection{DEFINITION OF $N_{sum}(t, a, b) \in \mathbb{R}$}

We define:
\begin{equation}
    N_{sum}(b, a, p, n) \coloneqq \sqrt{L(b, a)^p_{n}}
\end{equation}
If and only if $n_{min} < n < n_{max}, \quad n_{min} > 0$.
