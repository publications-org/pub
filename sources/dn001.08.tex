\section{DEFINITIONS}

We assume these definitions are valid within their conditions:

\subsection{DEFINITION OF A SUM $S(t) \in \mathbb{R}$}
We define:
\begin{equation}
    S(t, p) \coloneqq \sum_{k=1}^{p}\frac{t}{p}\cdot kt, \quad \int_{a}^{b}S(t, b - a)dt, \quad p \geq 1
\end{equation}
Where $\forall t \in \mathbb{R}, t > 1$.

\subsection{DEFINITION OF A SUM $W(t) \in \mathbb{R}$}
We define:
\begin{equation}
    W(t) \coloneqq S(t \cdot e)\cdot(\sum_{k=1}^{n} S((t+k)\cdot e)) +C_{rem}, \quad \forall C_{rem} \in \mathbb{R}, \quad C > 0
\end{equation}
Where $\forall t \in \mathbb{R}, t > 0$.

\subsection{DEFINITION OF THE NORM $N_{sum}(t) \in \mathbb{R}$}

We define:
\begin{equation}
    N_{sum}(t) \coloneqq \sqrt{S(t)^2_{n}} \quad \forall t \in \mathbb{R}
\end{equation}
If and only if $t > 1, \quad n_{min} < n < n_{max}, \quad n_{min} > 0$.
