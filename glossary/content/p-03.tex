\begin{theorem}
	Let the following:
\begin{equation}
    \mu(n, p) = \sum_{n}^{p} M(n, p); \quad \mu(n, p) \in \mathbb{D}^{+}
\end{equation}
There exists:
\begin{equation}
    \mu(n, p) = p! + \text{ } (n+1p)! + \text{ } (n+2p)! + \text{ } \dots \text{ } + (n+xp)!.
\end{equation}
\end{theorem}

\begin{theorem}
	Let for each rational number $x$ in $\mu(n, p)$ be within:
	\begin{align*}
		\frac{2\pi}{\mu(m, p)} < x < \mu(n, p).
	\end{align*}
\end{theorem}
