\section{Remarks as 'Ut est':}
%% ============================================================

\lettrine[lines=2, lhang=0.1, loversize=0.2]{T}he previous chapter states the following as both a remark and theorem:\\
Let us impose some remarks and definitions first.
\begin{remark}
	Per the glossary's [2.31] for $x = \frac{ln(\pi^\mathrm{e})!}{\pi}, \quad \forall x \in \mathbb{R}.$ We have the following:
	\begin{equation}
		\frac{ln(\pi^\mathrm{e})!}{\pi} = \frac{(\mathrm{e} \log(\pi))!}{\pi}, \quad x \neq 0.
	\end{equation}
\end{remark}

\begin{definition}
	Per [1.1]. Let us define the following $\forall x \in \mathbb{R}.$:
	\begin{align*}
		f(x) &= \int x \frac{ln(\pi^\mathrm{e})!}{\pi} dx.
	\end{align*}
	If and only If: $f(x), x \neq 0.$
\end{definition}
