\section{Theorems}
\begin{theorem}
    There exists in $\mathbb{R}$, the following identity:
    \begin{equation}
        \frac{\int (np) \text{ } dp}{\int (px) \text{ } dx} = \frac{p^2n+2C}{px^2+2C}
    \end{equation}
    which approaches zero for elements of $\mathbb{R}$ matching the following equation:
    \begin{equation}
        \lim_{x\to\infty} \frac{p^2n+2C}{px^2+2C} = 0
    \end{equation}
    such that $x \to \infty$, and $0 < x < \pi$.
\end{theorem}
