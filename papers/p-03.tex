\section{Theorems}

\begin{theorem}
For all returning elements in $F(w)$:
\begin{equation}
    F(w) = \sum_{i=1}^{n} w_i
\end{equation}
s.t.:
\begin{equation}
    w_{i}(n, m) = \sum_{j=n}^{m} w_{(i-1)-j}, \space \forall i > (n+2), \in \mathbb{N}
\end{equation}
for all $n, m \in \mathbb{N}$ and $n < m$.
\end{theorem}
