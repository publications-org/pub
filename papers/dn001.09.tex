\subsection{DEFINITION OF $C \in \mathbb{R}$:}
We define that for $C(n, p)$:
\begin{equation}
C(n, p) \coloneq \sum_{k=1}^{n} \sum_{d=1}^{p} I_{1} (k \cdot d) \in \mathbb{R}
\end{equation}
Where $p > S(x) > n$. If and only if $x \geq 0$.

\subsection{DEFINITION OF $I_{1} \in \mathbb{C}$:}
We define that for $I_{1}(t)$:
\begin{equation}
I_{1}(t) \coloneq \oint \frac{|Log(t\cdot z)|}{t}dz, \quad \forall z \in \mathbb{R}, \quad t > 0
\end{equation}
Where $Log$ is the Principal Complex Logarithm of: $t \cdot z$.
