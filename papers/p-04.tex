\section{Abstract}

\section{Introduction}

Defining the complexity of a task $t$ as $C(t)$, we can consider the space of complexities $S_{n}(C(t), p)$ for $n > 1, \space n \in \mathbb{N}$ and $p \in \mathbb{R}$.
\begin{equation}
  S_{n}(C(t), p) = \sum_{i=1}^{n} C(t){^{i}} \quad \text{for } p > 0
\end{equation}
Defining the following lemmas:

\section{Lemmas}
\begin{lemma}
If $C(t) > 0$, there exists a value of $t$ such that $t \in \mathbb{S_n}$.
\end{lemma}
\begin{lemma}
There exists a value of $C(t)$ such that $C(t) \in \mathbb{S_n}$.
\end{lemma}

\section{Theorems}
\begin{theorem}
If $C(t) > 0$, there exists a value of $t$ such
that $t \in \mathbb{S_n}$.
\end{theorem}

\begin{theorem}
  There exists:
  \begin{equation}
    S_{n}(C(t), p) = i^{p}
  \end{equation}
  For $p$ being a positive integer reaching $p$.
\end{theorem}

\begin{theorem}
  There exists:
  \begin{equation}
    \int_{C}^{\infty} \frac{\log\left(\left|\sum_{n=1}^{\pi} \frac{1}{n^s}\right|\right)}{\sum_{n=1}^{\pi} \frac{1}{n^s}} \, d\zeta(s) = \zeta(s) \quad \text{for } \text{Re}(s) > 1
  \end{equation}
  For $C > 1$.
\end{theorem}

\begin{theorem}
  There exists a value of $s$ between $\pi$ and $\lim_{n \to \infty} \frac{1}{n^s}$ such that:
  \begin{equation}
    \pi > \sum_{n=1}^{\pi} \frac{1}{n^s} > \lim_{n \to \infty} \frac{1}{n^s}
  \end{equation}
  For $s \in \mathbb{R}$. And $\lim_{n \to \infty} \frac{1}{n^s}$. This shall yield $s > 1.39425$.
\end{theorem}

\begin{theorem}
  There exists for $s > 1$ and s being a real number, the following:
  \begin{equation}
    \int_{C}^{\infty} \zeta(s) d \zeta(s) = \frac{Log(|\zeta(s)|)}{\zeta(s)}
  \end{equation}
\end{theorem}
