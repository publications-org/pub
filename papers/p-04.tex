\section{On Lemmas and Theorems}
\begin{lemma}
If $C(t) > 0$, there exists a value of $t$ such that $t \in \mathbb{S_n}$.
\end{lemma}
\begin{lemma}
There exists a value of $C(t)$ such that $C(t) \in \mathbb{S_n}$.
\end{lemma}

\begin{theorem}
  There exists:
  \begin{equation}
    S_{n}(C(t), p) = i^{p}
  \end{equation}
  For $p$ being a positive integer reaching $p$.
\end{theorem}

\begin{theorem}
  There exists an integral solution of:
  \begin{equation}
    \int \frac{\log\left(\left|\sum_{n=1}^{\pi} \frac{1}{n^s}\right|\right)}{\sum_{n=1}^{\pi} \frac{1}{n^s}} \, d \sum _{n=1}^{\pi }\:\frac{1}{n^s}
  \end{equation}
\end{theorem}

\begin{theorem}
  There exists a value of $s$ between $\pi$ and $\lim_{n \to \infty} \frac{1}{n^s}$ such that:
  \begin{equation}
    \pi > \sum_{n=1}^{\pi} \frac{1}{n^s} > \lim_{n \to \infty} \frac{1}{n^s}
  \end{equation}
  For $s \in \mathbb{R}$. And $\lim_{n \to \infty} \frac{1}{n^s}$ shall yield for $s > 1.39425$.
\end{theorem}

\begin{theorem}
  There exists for a definite integral that can be represented in the form of:
  \begin{equation}
    \int \sum_{n=1}^{\pi} \frac {1}{n^s} ds = \pi \frac{1-2^{-\pi}}{\log(2)}+\frac{1-3^{-\pi}}{\log(3)}
  \end{equation}
\end{theorem}
