	\section{Equations \& Theorems}
	
	\begin{definition}
		We define Euler–Mascheroni constant:
		\begin{equation}
			\gamma = \lim_{n \to \infty}(\sum_{k=1}^{n}(\frac{1}{k}-\log(n))).
		\end{equation}
		Which may be rewritten to per M. Ram Murty and N. Saradha findings:
		\begin{equation}
			\gamma(a, q) = \lim_{n \to \infty}\sum_{k=0}^{n}(\frac{1}{a+kq}-\frac{\log(a+nq)}{q}).
		\end{equation}
	\end{definition}
	
	\begin{definition}
		The following is the closed form of a series of low-integer-order polylogarithms:
		\begin{align*}
			\frac{\operatorname{d}}{\operatorname{dz}} \operatorname{Li}_{n}(z) &= \frac{\operatorname{Li}_{n-1}(z)}{z}.
		\end{align*}
	\end{definition}
	
	\begin{definition}
		The following is a series of the exponential function:
		\begin{align*}
			\sum_{k=0}^{\infty} \frac{z^k}{k!} &= e^{z}.
		\end{align*}
	\end{definition}
	
	\begin{theorem}
		The following is a series of Euler–Mascheroni constants over $k$:
		\begin{align*}
			\sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)}.
		\end{align*}
	\end{theorem}
	
	\begin{theorem}
		The following is a series of $k$ over Euler–Mascheroni constants:
		\begin{align*}
			\sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k}.
		\end{align*}
	\end{theorem}
	
	\begin{theorem}
		The following inequality may also be defined:
		\begin{align*}
			\sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} < x < \sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)}; \quad x \neq 0.
		\end{align*}
	\end{theorem}
	
	\begin{theorem}
		Applying the previous definition laid out on previous papers. We satisfy the following:
		\begin{align*}
			yt &= \pm T \frac{b}{c!}; \quad \pm T \in \mathbb{D}^{+}\\
			\frac{yt}{\pm T} &= c!; \quad \pm T \neq 0\\
			-c! &= \pm \frac{yt}{\pm T}.
		\end{align*}
		For any domain $\mathbb{D}^{+}$. Where the following is a Taylor series of:
		\begin{equation}
			T = \frac{x^n}{n!}; \quad n \neq 0.
		\end{equation}
	\end{theorem}
	
	\begin{proof}
		Assuming:
		\begin{align*}
			yt &= \sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} < x < \sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)};\\
			\frac{yt}{\pm T} &= \frac{\sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} < x < \sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)}}{\pm T}.
		\end{align*}
		Now in order to continue further, we need to solve:
		\begin{align*}
			\sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} &\leq x < \sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)};\\
			\sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} &\leq x - \sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)} \geq 0;\\
			\sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} + \sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)} &\geq x;
		\end{align*}
		Therefore. For $x = \sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} + \sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)}.$
		\begin{align*}
			\pm T + yt &= \sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} + \sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)};\\
			yt &= \frac{\sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} + \sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)}}{\pm T}.
		\end{align*}
		Thus. For: x = $\sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} + \sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)}.$:
		\begin{align*}
			\pm T + \frac{\sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} + \sum_{k=1}^{\infty}\frac{k}{\gamma(a, qk)}}{\pm T} &= \sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} + \sum_{k=1}^{\infty}\frac{k}{\gamma(a, qk)};\\
			\frac{\sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} + \sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)}}{\pm T} &= \frac{\sum_{k=1}^{\infty} \frac{\gamma(a, qk)}{k} + \sum_{k=1}^{\infty} \frac{k}{\gamma(a, qk)}}{\pm T}; \quad \pm T \neq 0.
		\end{align*}
	\end{proof}
