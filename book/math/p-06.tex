\section{Equations \& Theorems}

\begin{theorem}
  Let $pw(n, p, a) \in \mathbb{R}$, for $p > 0$, $x > 0$, per [1.1]
  \begin{equation}
    pw(n, p, a) = \int_D \frac{a(n)}{p} da
  \end{equation}
  That integrates to:
  \begin{equation}
    PW(n, p, a) = \int_D \frac{a(n)}{p} da = \frac{n}{p} \frac{a^2}{2}\pm C
  \end{equation}
  We will then form the following identities by dviding $-PW(n, p, a)$.
  \begin{equation}
    \pm D = \frac{\pm \frac{n}{p} \frac{a^2}{2}\pm C}{\pm PW(n, p, a)}
  \end{equation}
  Therefore:
  \begin{equation}
   	\frac{\pm D}{\pm C} = \frac{\pm \frac{n}{p} \frac{a^2}{2}}{\pm \frac{n}{p} \frac{a^2}{2}\pm C}
  \end{equation}
  Which then becomes:
  \begin{equation}
   	\frac{\pm D}{\pm C \frac{n}{p} \frac{a^2}{2}} = K; \quad K = \frac{1}{\pm C};
  \end{equation}
\end{theorem}

\begin{theorem}
  For the following:
  \begin{equation}
    \frac{dy}{\operatorname{dx}}pw(n, p, a) = v_{n}; \quad v_{n} \in \mathbb{R}; \quad pw(n, p, a) \in \mathbb{R}
  \end{equation}
  There is:
  \begin{equation}
    v_{n} \in \mathbb{R}; \quad v_{n}\neq 0
  \end{equation}
\end{theorem}

\begin{theorem}
  There exists:
  \begin{equation}
    cp = \frac{p!}{\ln \left(p!\right)}
  \end{equation}
  For:
  \begin{equation}
    c = \frac{p!}{p\ln \left(p!\right)}; \quad p \neq 0
  \end{equation}
\end{theorem}

\begin{theorem}
  For:
  \begin{equation}
    \int_D \frac{p!}{p\ln \left(p!\right)!}\operatorname{dp} = z
  \end{equation}
  Then.
  \begin{equation}
    -z = \frac{1}{\int_D \frac{p!}{p\ln \left(p!\right)}\operatorname{dp}}
  \end{equation}
  Therefore:
  \begin{equation}
    \frac{1}{\int_D \frac{p!}{p\ln \left(p!\right)}\operatorname{dp}} < \int_D \frac{p!}{p\ln \left(p\right)!}\operatorname{dp}; \quad z \neq 0; \quad z > -z
  \end{equation}
  There exists an identity of the above. For $z \in \mathbb{R}$.
\end{theorem}
