\section{Equations \& Theorems}

\begin{definition}
	Let the following be defined for any domain D:
\begin{align*}
	x + F(n, p, s, x) &= \frac{\int (\frac{p}{n^{s}}) \text{ } \operatorname{dx}}{\frac{p}{n^{s}}! };\\
	x (n^{s}!) + F(n, p, s, x)(n^{s}!) &= \frac{ {\int (\frac{p}{n^{s}}) \text{ } \operatorname{dx}}{p!}}{p! (n^{s}!)};\\
	x (n^{s}!) &= \frac{ {\int (\frac{p}{n^{s}}) \text{ } \operatorname{dx}}{p!}}{p! (n^{s}!)} - F(n, p, s, x)(n^{s}!);\\
		x (n^{s}!) &= \int (\frac{p}{n^{s}}) \text{ } \operatorname{dx}{p!} - F(n, p, s, x)(n^{s}!) p! (n^{s}!);\\\\
	x (n^{s}!) -\int (\frac{p}{n^{s}}) \text{ } \operatorname{dx} &= p! - F(n, p, s, x)(n^{s}!) p! (n^{s}!);\\
	-\int (\frac{p}{n^{s}}) \text{ } \operatorname{dx} &= p! - F(n, p, s, x)(n^{s}!) p! (n^{s}!) - x (n^{s}!);\\
	-\int (\frac{p}{n^{s}})\text{ } \operatorname{dx} &= p! - F(n, p, s, x)p!(n^{s}!)^{2} - x (n^{s}!);\\
	-\int (\frac{p}{n^{s}})\text{ } \operatorname{dx} &= - F(n, p, s, x)2p!(n^{s}!)^{2} - x (n^{s}!) \in \mathbb{D}.
\end{align*}
\end{definition}
