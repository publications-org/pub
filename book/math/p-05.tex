\section{Equations \& Theorems}

\begin{definition}
	\begin{align*}
		\pm B &= \frac{d}{dx} \frac{x}{n-\ln (p!)+x}\\
		0 &= \frac{d}{dx} \frac{x}{n-\ln (p!)+x} - B
	\end{align*}
\end{definition}

\begin{theorem}
Assume $\rho(n, p, s, x) \in \mathbb{R}$:
\begin{align*}
\pm C + \rho(n, p, s, x) &= \frac{\int (\frac{p}{n^{s}}) \text{ } dx}{\frac{p}{n^{s}}! }\\
\pm C &= \frac{ \frac{\int (\frac{p}{n^{s}}) \text{ } dx}{\frac{p}{n^{s}}! } }{\rho(n, p, s, x)}
\end{align*}
Assuming $x = \pi$:
\begin{equation}
  \lim_{x \to \pi} \rho(n, p, s, x) \to K; \quad K = \pm \frac{x p\pm Cn^s}{n^s\frac{p}{n^s}!}
\end{equation}
If $x > 0$. And $\lim x \to \pi$. Therefore:
\begin{align*}
	\pm K \pm C &= \pm Z\\
	\pm K &= \frac{\pm Z}{\pm C}
\end{align*}
\end{theorem}
