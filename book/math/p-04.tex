\section{Equations \& Theorems}

\begin{definition}
	There exists an identity of a function $\lambda(x)$:
	\begin{align*}
		\pm y \lambda(x) &= \frac{\operatorname{d}}{\operatorname{dx}} \frac{P'(x)}{\pm P(x)}; \quad \pm P(s) \neq 0.
	\end{align*}
\end{definition}

\begin{definition}
	There exists an equation $\delta$:
	\begin{align*}
		\delta &= \frac{\operatorname{d}}{\operatorname{dx}} \frac{g(x) + h(x)}{\gamma(x)}.
	\end{align*}
	Thus, let the following equation:
	\begin{align*}
		\frac{\operatorname{d}}{\operatorname{dx}}f(x) + \lambda(x) &= \frac{\operatorname{d}}{\operatorname{dx}} \frac{g(x) + h(x)}{\gamma(x)} \delta; \quad \gamma(x) \neq 0.
	\end{align*}
\end{definition}
