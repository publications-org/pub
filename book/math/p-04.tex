\section{Equations \& Definitions}

\begin{definition}
	Let the following:
	\begin{align*}
		E(x) &= \frac{\gamma(x)}{\delta(x) + \lambda(x)}; \quad \gamma(x) \neq 0.
	\end{align*}
\end{definition}

\begin{definition}
	Let the following based on $E(x)$:
	\begin{align*}
		P' &= \prod_{n=1}^{3}\rho(x) x^{n} + \int E(x) dx;\\
		P' &= \prod_{n=1}^{3}\rho(x) x^{n} + \int \frac{\gamma(x)}{\delta(x) + \lambda(x)} dx; \quad P(s) \neq 0.
	\end{align*}
\end{definition}

\begin{definition}
	The following based for $x \in \mathbb{N}$:
	\begin{align*}
		\prod_{n=1}^{3}\rho(x) x^{n} + \int \frac{\gamma(x)}{\delta(x) + \lambda(x)} dx; \quad P(s) \neq 0.
	\end{align*}
	Admits a solution in $\mathbb{R}$ of the following:
	\begin{align*}
		P'' &= \prod_{n=1}^{3}\rho(x) x^{n} + \int \frac{\gamma(x)}{\delta(x) + \lambda(x)} dx; \quad P'' \neq 0.
	\end{align*}
\end{definition}

\begin{definition}
	Let the following:
	\begin{align*}
		&= \frac{\gamma(x)}{\delta(x)};\\
	 	\lambda(x) &=  \lambda(x) \frac{\gamma(x)}{\delta(x)};\\
	 	L' &= \lambda(x) \frac{\gamma(x)}{\delta(x)} - \lambda(x); \quad \delta(x) \neq 0.
	\end{align*}
\end{definition}

\begin{definition}
	We must find a value of $\ln x$ that equals:
	\begin{align*}
		&= \frac{\ln x}{\Delta y};\\
		\Delta y &= \ln x;
	\end{align*}
\end{definition}

\begin{definition}

	\begin{align*}
		r &= ( \frac{\ln x}{\Delta n} + \frac{\Delta n}{\ln x})^m; \quad r \neq 0; \quad r \in \mathbb{R}{+};\\
		I' &= - r + ( \frac{\ln x}{\Delta n} + \frac{\Delta n}{\ln x})^m; \quad r \neq 0; \quad r \in \mathbb{R}{+};
