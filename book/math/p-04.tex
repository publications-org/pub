\section{Equations \& Theorems}
\begin{theorem}
  There exists for $h(n, s)$ a solution of:
  \begin{align*}
    h(n, s) &= \oint_{C} \frac{\ln(Y)}{Y} = -n^s s+\frac{n^s}{\ln \left(n\right)}\pm C; \quad Y = \frac{1}{n^s};\\
     \pm D &= \lim _{x\to \infty } -n^s s+\frac{n^s}{\ln \left(n\right)}\pm C;\\
     \frac{\pm D}{\pm C} &= \frac{\lim _{x\to \infty } -n^s s+\frac{n^s}{\ln \left(n\right)}}{\pm C};
  \end{align*}
\end{theorem}

\begin{theorem}
  There exists for $z$:
  \begin{equation}
    z = \lim _{n\:\to \pi }\:(S)^{n}; \quad S = \sum_{d=1}^{p} \frac{d}{^p}
  \end{equation}
  Therefore:
  \begin{equation}
    z \to \pm \left(\frac{p+1}{2}\right)^{\pi }
  \end{equation}
  If and only if $y \in \mathbb{R}$, $y > 0$. And if and only if: $p \in \mathbb{R}$.
\end{theorem}

\begin{theorem}
  There exists for $\operatorname{eq}(x)$
  \begin{equation}
    \operatorname{eq}(x) = \pm \left(\frac{p+1}{2}\right)^{x}
  \end{equation}
  If and only If:
  \begin{align*}
    \lim_{x \to \operatorname{e}} \pm \operatorname{eq}(x) = \operatorname{eq}(x)\\
    \frac{ \lim_{x \to \operatorname{e}} \pm \operatorname{eq}(x)}{\operatorname{eq}(x)} = \frac{\operatorname{eq}(x)}{\operatorname{eq}(x)} = \frac{1}{1}
  \end{align*}
\end{theorem}

\begin{theorem}
  For $\sigma(x, y, z)$. Then
  \begin{equation}
    \sigma(x, y, z) = \int \frac{f'(x y z)}{f''(x y z)}
  \end{equation}
  If and only If:
  \begin{equation}
    f\:''\left(x\right)= f\:'\left(x\right)\pm C=f\left(x\right)\pm C \pm D=\frac{1}{x} \pm C \pm D \pm E
  \end{equation}
  There exists for:
  \begin{equation}
    f\left(x\right)\:\ne 0
  \end{equation}
  That equals:
  \begin{equation}
    C = (\sigma(x, y, z))^{2}
  \end{equation}
  For:
  \begin{equation}
    \frac{C}{\sigma(x, y, z)} = \sigma(x, y, z)
  \end{equation}
\end{theorem}

\begin{theorem}
  There exists:
  \begin{equation}
    \tau (s) \coloneq \lim _{s\to \:\pi } -n^s+\frac{n^s}{\ln \left(n\right)}\pm C
  \end{equation}
  If and only If: $\tau (s) \in \mathbb{R}$
\end{theorem}

\begin{theorem}
  For:
  \begin{equation}
    \alpha(a, \dots, z) = \int \lambda(a, \dots, z)! 
  \end{equation}
  If and only If:
  \begin{equation}
    \lim_{x \to 0} \alpha(a, \dots, z) = +\infty
  \end{equation}
\end{theorem}
