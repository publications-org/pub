\section{Equations \& Theorems}
\begin{theorem}
    There exists in $\mathbb{R}$:
    \begin{equation}
        \frac{\int (np) \text{ } dp}{\int (px) \text{ } dx} = \frac{p^2n+2C}{px^2+2C}
    \end{equation}
    \begin{equation}
        \lim_{x\to\infty} \frac{p^2n+2C}{px^2+2C} = \frac{\operatorname{undefined}}{c} = C
    \end{equation}
\end{theorem}
