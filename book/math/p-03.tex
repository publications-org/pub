\section{Equations \& Theorems}

\begin{theorem}
For:
\begin{equation}
    A(n, p) \coloneq \sum_{n}^{p} M(n p)
\end{equation}
\begin{equation}
    A(n, p) \implies p! + \text{ } (n+1p)! + \text{ } (n+2p)! + \text{ } \dots \text{ } + (n+Lp)! 
\end{equation}
For $A(n, p) \neq 0$, the previous identity holds for numbers in $\mathbb{R}$.
\end{theorem}
